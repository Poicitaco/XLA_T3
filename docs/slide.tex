\section{Detection \& Recognition}

% ---- Frame 5: YOLO ----
\begin{frame}[fragile]{YOLOv11n-face --- Phát Hiện Khuôn Mặt}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{Tại sao YOLOv11n?}
\begin{itemize}\footnotesize
  \item \textbf{Nano}: tối ưu inference trên CPU
  \item Fine-tuned đặc biệt cho face detection
  \item Hỗ trợ NMS (Non-Max Suppression)
  \item conf\_threshold = \textbf{0.35}, iou\_threshold = \textbf{0.50}
\end{itemize}
\end{block}

\begin{block}{Optimize: detect\_every}
\footnotesize
Frame xen kẽ: chỉ gọi YOLO mỗi \textbf{2 frame}.\\
Frame bỏ qua dùng lại kết quả lần trước.\\
$\Rightarrow$ Giảm 50\% compute, lag không đáng kể.
\end{block}

\column{0.46\textwidth}
\begin{lstlisting}[language=Python, caption={}]
class FaceDetector:
  def detect_faces(self, frame):
    results = self.model(
      frame,
      imgsz=512,       # inference size
      conf=0.35,       # min confidence
      iou=0.50,        # NMS threshold
      verbose=False,
    )
    boxes = []
    for r in results:
      for box in r.boxes:
        x1,y1,x2,y2 = map(int,
          box.xyxy[0])
        boxes.append(
          [x1, y1,
           x2-x1, y2-y1])
    return boxes
\end{lstlisting}
\end{columns}
\end{frame}

% ---- Frame 6: InsightFace ----
\begin{frame}[fragile]{InsightFace Buffalo-L --- Nhận Diện Thành Viên}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{Embedding 512 chiều}
\footnotesize
\begin{itemize}
  \item Model \textbf{buffalo\_l}: state-of-the-art recognition
  \item Vector 512 chiều = ``vân tay số'' khuôn mặt
  \item Cosine similarity: $-1$ (hoàn toàn khác) đến $+1$ (giống hệt)
  \item Threshold: \textbf{0.35}
  \begin{itemize}\scriptsize
    \item $ > 0.35$ $\Rightarrow$ thành viên $\Rightarrow$ \textcolor{darkgreen}{\textbf{không blur}}
    \item $\leq 0.35$ $\Rightarrow$ người lạ $\Rightarrow$ \textcolor{accent}{\textbf{blur}}
  \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Vectorized Matching}
\footnotesize
So sánh \textit{một lần} với \textbf{toàn bộ} embedding matrix:\\
$$\text{sim}(a,b) = \frac{a \cdot b}{\|a\| \cdot \|b\|}$$
$\Rightarrow$ O(n) không cần vòng lặp Python
\end{block}

\column{0.46\textwidth}
\begin{lstlisting}[language=Python, caption={}]
class FaceRecognizer:
  def __init__(self):
    self.app = FaceAnalysis(
      name="buffalo_l",
      providers=[
        "CPUExecutionProvider"
      ]
    )
    self.app.prepare(
      ctx_id=0,
      det_size=(256, 256)
    )

  @staticmethod
  def cosine_similarity(a, b):
    return float(
      np.dot(a, b) / (
        np.linalg.norm(a)
        * np.linalg.norm(b)
      )
    )
\end{lstlisting}
\end{columns}
\end{frame}

def encrypt_clip(frames, fps, password):
    # Step 1: Serialize frames
    plaintext = _pack_frames(frames, fps)
    # [4B n_frames][4B fps*1000]
    # for each: [4B len][JPEG bytes]

    # Step 2: Random per-clip salt
    salt  = secrets.token_bytes(32)  # 256-bit
    nonce = secrets.token_bytes(12)  # 96-bit GCM

    # Step 3: Key derivation
    key = derive_clip_key(password, salt)
    # PBKDF2-SHA256, 390000 rounds -> 32B

    # Step 4: AES-256-GCM encrypt
    ct = AESGCM(key).encrypt(
        nonce, plaintext, None)
    # Output includes 128-bit Auth Tag

    # Step 5: Pack file
    return salt + nonce + ct
    # [32B salt][12B nonce][ciphertext]
\end{lstlisting}

\column{0.44\textwidth}
\begin{lstlisting}[language=Python, caption={src/clip\_recorder.py}]
def decrypt_clip(data, password):
    if len(data) < 44:
        raise ValueError("Corrupted")

    # Unpack file format
    salt      = data[:32]
    nonce     = data[32:44]
    ciphertext = data[44:]

    # Re-derive same key
    key = derive_clip_key(password, salt)

    try:
        # Decrypt + verify Auth Tag
        plaintext = AESGCM(key).decrypt(
            nonce, ciphertext, None)
    except Exception:
        raise ValueError(
            "Decryption failed - "
            "wrong password or "
            "corrupted clip.")

    return _unpack_frames(plaintext)
\end{lstlisting}

# OWASP 2023 minimum
_ITERATIONS = 390_000
_KEY_LEN = 32  # 256-bit

def _pbkdf2(password, salt,
            iterations=_ITERATIONS):
    kdf = PBKDF2HMAC(
        algorithm=hashes.SHA256(),
        length=_KEY_LEN,
        salt=salt,
        iterations=iterations,
    )
    return kdf.derive(
        password.encode("utf-8"))

def _ct_equal(a: bytes, b: bytes):
    """Prevent Timing Attack."""
    # Python '==' stops at first diff byte
    # Attacker measures response time
    # -> can guess chars one by one
    # hmac.compare_digest: CONSTANT TIME
    return hmac.compare_digest(a, b)

  """4-step pipeline, IRREVERSIBLE"""

  # Step 1: Compress 8% resolution
  # Destroy all facial details
  w, h = roi.shape[:2]
  small = cv2.resize(roi,
    (int(w*0.08), int(h*0.08)),
    interpolation=INTER_AREA)
  work = cv2.resize(small, (w, h),
    interpolation=INTER_NEAREST)

  # Step 2: Scramble 12x12 blocks
  # Break geometric relationships
  work = _scramble_blocks(work,
    block_size=12, seed=1337)

  # Step 3: Blur edges (kernel 35)
  work = cv2.GaussianBlur(work,
    (35,35), 0)

  # Step 4: Quantize to 10 colors
  # Remove subtle texture traces
  work = _quantize_colors(work,
    levels=10)
  return work
